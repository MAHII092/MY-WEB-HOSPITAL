📋 Setup Instructions
  1. Google Sheets मध्ये नवीन spreadsheet बनवा
  2. Tools → Script Editor उघडा
  3. खालील Apps Script code पेस्ट करा
  4. Deploy → New Deployment → Web App म्हणून publish करा
  5. Generated URL इथे paste करा
function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['नाव','मोबाइल','ईमेल','वय','लिंग',
      'विभाग','डॉक्टर','तारीख','वेळ','संदेश','वेळ_नोंद','Client']);
  }
  
  sheet.appendRow([
    data['नाव'] || '', data['मोबाइल'] || '',
    data['ईमेल'] || '', data['वय'] || '',
    data['लिंग'] || '', data['विभाग'] || '',
    data['डॉक्टर'] || '', data['तारीख'] || '',
    data['वेळ'] || '', data['संदेश'] || '',
    data['वेळ_नोंद'] || '', data['client'] || ''
  ]);
  
  return ContentService.createTextOutput('OK');
}
